	teleportButton.Visible = false
end)

	frameGeral.Visible = false
end)

	wait(0.001)
	game.Players.LocalPlayer.Character:ScaleTo(tamanhoGigante.Value)
	game.Players.LocalPlayer.Character.Humanoid.JumpHeight = 7.2
end)
